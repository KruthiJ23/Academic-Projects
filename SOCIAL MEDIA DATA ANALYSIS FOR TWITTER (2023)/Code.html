#Calculating the modularity and plotting the network graph
grp = max((G.subgraph(c) for c in nx.connected_components(G)), key=len)

partition = community.best_partition(grp)
modularity = community.modularity(partition, grp)
print('Modularity:', modularity)

colors = [partition[n] for n in grp.nodes()]
my_colors = plt.cm.plasma
pos = nx.spring_layout(grp, seed=10396950)
nx.draw_networkx_nodes(grp, pos, node_color=colors, cmap = my_colors,  node_size=10)
nx.draw_networkx_edges(grp, pos, alpha=0.5)

plt.show()
